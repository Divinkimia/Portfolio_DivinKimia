FS Barber

Salon mixte Hommes, Femmes & Enfants. Prestations, fidélité, boutique, rendez-vous en ligne.

Vue d’ensemble

Présentation du projet

FS Barber est un site web complet pour salon de coiffure : galerie de coupes (hommes/femmes), programme de fidélité avec points, boutique en ligne avec panier et paiement Stripe, prise de rendez-vous, espace client et back-offices admin / superAdmin.

  • PHP
  • MySQL
  • Stripe

Périmètre fonctionnel

Fonctionnalités principales

Quatre blocs métier, mis en avant dans une grille type « bento ».

01

Coupes & prestations

Galerie coupes hommes/femmes, tarifs et prestations.

Rendez-vous

Réservation en ligne, planning coiffeurs et créneaux disponibles en temps réel.

Les parties du site

Chaque espace du projet a son rôle : accueil, prestations, boutique, rendez-vous, espace client et back-office.

Accueil

Introduction immersive, vidéo de présentation du salon, galeries de coupes hommes & femmes, un aperçu de la fidélité, de la boutique, et des avis certifiés.

Prestations

Catalogue clair avec tarification dynamique structurée par catégories (Homme, Femme, Enfant, Soin). Design épuré pour mettre en valeur le savoir-faire.

Boutique & Paiement

Un e-commerce complet intégré. Gestion du panier asynchrone et processus de checkout fluide 100% sécurisé via l'API Stripe.

Prise de Rendez-vous

Une interface de réservation intuitive. Choix du coiffeur, sélection des prestations et affichage intelligent des créneaux disponibles en temps réel.

Espace Client

Un dashboard personnalisé : suivi des commandes, historique des visites, gestion du profil et affichage en un coup d'œil du solde de points de fidélité.

Administration

Des back-offices puissants (Admin et SuperAdmin) pour garder le contrôle : gestion du planning, des plannings d'équipe, des produits et des réservations.

-- Table clients (fidélité, coordonnées)
CREATE TABLE `clients` (
  `id_client` int(11) NOT NULL AUTO_INCREMENT,
  `num_client` varchar(12) DEFAULT NULL,
  `nom` varchar(100) NOT NULL,
  `prenom` varchar(100) NOT NULL,
  `email` varchar(150) NOT NULL,
  `telephone` varchar(20) NOT NULL,
  `fidelite_points` int(11) NOT NULL DEFAULT 100,
  `date_inscription` timestamp NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`id_client`),
  UNIQUE KEY `email` (`email`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Table users (connexion site + lien client)
CREATE TABLE `users` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `email` varchar(255) NOT NULL,
  `password` varchar(255) NOT NULL,
  `first_name` varchar(100) DEFAULT NULL,
  `last_name` varchar(100) DEFAULT NULL,
  `role` enum('client','admin') DEFAULT 'client',
  `id_client` int(11) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `email` (`email`),
  KEY `id_client` (`id_client`),
  CONSTRAINT `users_ibfk_1` FOREIGN KEY (`id_client`) REFERENCES `clients` (`id_client`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Table rendezvous
CREATE TABLE `rendezvous` (
  `id_rdv` int(11) NOT NULL AUTO_INCREMENT,
  `id_client` int(11) NOT NULL,
  `id_coiffeur` int(11) NOT NULL,
  `id_service` int(11) NOT NULL,
  `date_rdv` date NOT NULL,
  `heure_rdv` time NOT NULL,
  `statut` enum('En attente','Confirmé','Terminé','Annulé') DEFAULT 'En attente',
  PRIMARY KEY (`id_rdv`),
  KEY `id_client` (`id_client`),
  KEY `id_coiffeur` (`id_coiffeur`),
  KEY `id_service` (`id_service`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Table commandes (boutique)
CREATE TABLE `commandes` (
  `id_commande` int(11) NOT NULL AUTO_INCREMENT,
  `id_client` int(11) NOT NULL,
  `total` decimal(10,2) NOT NULL,
  `statut` enum('En attente','Payée','Expédiée','Annulée') DEFAULT 'En attente',
  PRIMARY KEY (`id_commande`),
  KEY `id_client` (`id_client`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
PHP config.php — PDO getDBConnection(), constantes BDD
PHP 
// config.php - Connexion base de données
define('DB_HOST', '127.0.0.1');
define('DB_NAME', 'fs_barber');
define('DB_USER', 'root');
define('DB_PASS', '');
define('DB_CHARSET', 'utf8mb4');

function getDBConnection() {
    static $pdo = null;
    if ($pdo !== null) return $pdo;

    try {
        $dsn = "mysql:host=" . DB_HOST . ";dbname=" . DB_NAME . ";charset=" . DB_CHARSET;
        $options = [
            PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
            PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
            PDO::ATTR_EMULATE_PREPARES => false,
        ];
        $pdo = new PDO($dsn, DB_USER, DB_PASS, $options);
        return $pdo;
    } catch (PDOException $e) {
        error_log("Erreur BDD: " . $e->getMessage());
        return null;
    }
}

// includes/session.php
if (session_status() !== PHP_SESSION_ACTIVE) {

function fsb_nav_context(): array {
    $loggedIn = !empty($_SESSION['logged_in']) || !empty($_SESSION['user_id']);
    $points = isset($_SESSION['fidelite_points']) ? (int) $_SESSION['fidelite_points'] : 0;
    $displayName = '';

    if ($loggedIn && !empty($_SESSION['user_last_name']) && !empty($_SESSION['user_first_name'])) {
        $lastName = trim((string) $_SESSION['user_last_name']);
        $firstName = trim((string) $_SESSION['user_first_name']);
        $initial = $lastName !== '' ? mb_strtoupper(mb_substr($lastName, 0, 1), 'UTF-8') : '';
        $normalizedFirst = $firstName !== '' ? mb_strtolower($firstName, 'UTF-8') : '';
        $displayName = $initial !== '' ? sprintf('%s.%s', $initial, $normalizedFirst) : $normalizedFirst;
    }

    return [
        'logged_in' => $loggedIn,
        'display_name' => $displayName,
        'points' => $points,
    ];
}

function get_base_path() {
    $script_path = $_SERVER['SCRIPT_NAME'] ?? $_SERVER['PHP_SELF'] ?? '';
    $script_path = str_replace('\\', '/', $script_path);
    $base_path = dirname($script_path);
    $base_path = str_replace('\\', '/', $base_path);

    if (strpos($script_path, '/PHP/') !== false || basename(dirname($script_path)) === 'PHP') {
        return '../';
    }
    if (in_array(end(explode('/', trim($base_path, '/'))), ['admin', 'superAdmin'])) {
        return '../../';
    }
    return '';
}
